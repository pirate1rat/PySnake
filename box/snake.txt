import snake as sk
import pygame
import sys
import copy
import random
from enum import Enum

BLOCK_SIZE = int(20) #in px
WIDTH = 20 #in blocks
HIGHT = 20
GAME_SPEED = 200
count = 0

GREEN = (0, 255, 0)
GREY = (69, 67, 61)
RED = (255, 0, 0)
BLACK = (0, 0, 0)

res = (WIDTH*BLOCK_SIZE, HIGHT*BLOCK_SIZE)
screen = pygame.display.set_mode(res)
board = [[1 for _ in range(0, WIDTH)] for _ in range(0, HIGHT)]
player = sk.Snake(WIDTH//2, HIGHT//2, HIGHT)
apple = pygame.Vector2(random.randint(1, WIDTH - 2), random.randint(1, HIGHT - 2))

class Tile(Enum):
    LIMIT = 0
    NORMAL = 1
    SNAKE = 2
    APPLE = 3

def Initz()->None:
    for x in range(0, WIDTH):
        board[x][0] = 0
        board[x][HIGHT - 1] = 0
    for y in range(0, HIGHT):
        board[0][y] = 0
        board[WIDTH - 1][y] = 0
    
    for segment in player.body:
        board[int(segment.x)][int(segment.y)] = Tile.SNAKE.value
    
    board[int(apple.x)][int(apple.y)] = Tile.APPLE.value

def UpdateGame() -> bool:
    global count
    count += 1
    
    #keyboard input
    keys = pygame.key.get_pressed()
    new_movec = pygame.Vector2(0, 0)
    if keys[pygame.K_s]:
        new_movec = pygame.Vector2(0, 1)
    if keys[pygame.K_w]:
        new_movec = pygame.Vector2(0, -1)
    if keys[pygame.K_d]:
        new_movec = pygame.Vector2(1, 0)
    if keys[pygame.K_a]:
        new_movec = pygame.Vector2(-1, 0)

    if count == GAME_SPEED:
        count = 0

        if new_movec != pygame.Vector2(0, 0) and player.movec != -new_movec:
            player.movec = copy.copy(new_movec)

        player.head += player.movec

        if board[int(player.head.x)][int(player.head.y)] == Tile.LIMIT.value:
            return False
        elif board[int(player.head.x)][int(player.head.y)] == Tile.SNAKE.value:
            return False

        if board[int(player.head.x)][int(player.head.y)] == Tile.APPLE.value:
            player.body.insert(0, copy.copy(player.head))
            board[int(player.head.x)][int(player.head.y)] = Tile.SNAKE.value
            while True:
                apple = pygame.Vector2(random.randint(1, WIDTH - 2), random.randint(1, HIGHT - 2))
                if board[int(apple.x)][int(apple.y)] == Tile.NORMAL.value:
                    board[int(apple.x)][int(apple.y)] = Tile.APPLE.value
                    break
        else:
            player.body.insert(0, copy.copy(player.head))
            board[int(player.head.x)][int(player.head.y)] = Tile.SNAKE.value
            old_poz = player.body.pop()
            board[int(old_poz.x)][int(old_poz.y)] = Tile.NORMAL.value
    

    return True

def ComposeFrame() -> None:
    #drawing
    screen.fill(BLACK)
    for x in range(0, WIDTH):
        for y in range(0, HIGHT):
            match board[x][y]:
                case Tile.LIMIT.value:
                    pygame.draw.rect(screen, GREY, (x * BLOCK_SIZE, y * BLOCK_SIZE, BLOCK_SIZE, BLOCK_SIZE))
                case Tile.APPLE.value:
                    pygame.draw.rect(screen, RED, (x * BLOCK_SIZE, y * BLOCK_SIZE, BLOCK_SIZE, BLOCK_SIZE))
                case Tile.SNAKE.value:
                    pygame.draw.rect(screen, GREEN, (x * BLOCK_SIZE, y * BLOCK_SIZE, BLOCK_SIZE, BLOCK_SIZE))
                case _:
                    pass
    pygame.display.flip()

#main
Initz()
while True:

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            sys.exit(0)
    
    if not UpdateGame():
        break
    ComposeFrame()
